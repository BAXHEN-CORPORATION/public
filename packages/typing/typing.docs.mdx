import { Typing } from './typing';
import Typography from '@mui/material/Typography';

A component that simulates typing of texts passed as props.

### Component usage

```js
<Typing texts={['Youtuber', 'Designer', 'Programmer']} />
```

### Simple example!

```js live
<Typography variant="h3">
  I am a
  <Typing
    texts={['Youtuber', 'Designer', 'Programmer']}
    infinite
    speed={300}
    variant="h3"
  />
</Typography>
```
