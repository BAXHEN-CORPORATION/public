import { FancyUpload } from "./fancy-upload";

A component to go throut the flow of uploading a single file display the task progress and error or success state

### Component usage

```js
<FancyUpload
  onUpload={() => {}}
  onDone={() => {}}
  onCopy={() => {}}
  progress={0}
/>
```

### Render hello world!

```js live
<div style={{ backgroundColor: "gray", padding: "2rem" }}>
  <FancyUpload
    onUpload={(file, updateStatus) => {
      updateStatus("success");
    }}
    onDone={() => {}}
    onCopy={() => {}}
    progress={50}
  />
</div>
```
